 724/**
 725 * @brief Inserts a key and recipe into the internal nodes of the B+ tree.
 726 *
 727 * @param key The key to insert.
 728 * @param recipe Pointer to the recipe to insert.
 729 * @param cursor Pointer to the current B+ tree node.
 730 * @param root Pointer to the root of the B+ tree.
 731 */
 1732void insertInternal(int key, Recipe* recipe, struct BPlusTreeNode* cursor, struct BPlusTreeNode** root) {
 733  if (cursor->keyCount < MAX_RECIPES) {
 734    // Insert into non-full node
 735    int i = cursor->keyCount - 1;
 736    while (i >= 0 && cursor->keys[i] > key) {
 737      cursor->keys[i + 1] = cursor->keys[i];
 738      cursor->recipes[i + 1] = cursor->recipes[i];
 739      i--;
 740    }
 741    cursor->keys[i + 1] = key;
 742    cursor->recipes[i + 1] = recipe;
 743    cursor->keyCount++;
 744  }
 745  else {
 746    // Split full node
 747    int tempKeys[MAX_RECIPES + 1];
 748    Recipe* tempRecipes[MAX_RECIPES + 1];
 749    for (int i = 0; i < MAX_RECIPES; ++i) {
 750      tempKeys[i] = cursor->keys[i];
 751      tempRecipes[i] = cursor->recipes[i];
 752    }
 753
 754    int i = MAX_RECIPES - 1;
 755    while (i >= 0 && tempKeys[i] > key) {
 756      tempKeys[i + 1] = tempKeys[i]; tempRecipes[i + 1] = tempRecipes[i]; i--;
 757    }
 758    tempKeys[i + 1] = key;
 759    tempRecipes[i + 1] = recipe;
 760
 761    struct BPlusTreeNode* newLeaf = createNode(1);
 762    cursor->keyCount = (MAX_RECIPES + 1) / 2;
 763    newLeaf->keyCount = MAX_RECIPES + 1 - cursor->keyCount;
 764
 765    for (int i = 0; i < cursor->keyCount; ++i) {
 766      cursor->keys[i] = tempKeys[i];
 767      cursor->recipes[i] = tempRecipes[i];
 768    }
 769    for (int i = 0; i < newLeaf->keyCount; ++i) {
 770      newLeaf->keys[i] = tempKeys[cursor->keyCount + i];
 771      newLeaf->recipes[i] = tempRecipes[cursor->keyCount + i];
 772    }
 773
 774    newLeaf->next = cursor->next;
 775    cursor->next = newLeaf;
 776
 777    // Insert new key to parent
 778    if (cursor == *root) {
 779      struct BPlusTreeNode* newRoot = createNode(0);
 780      newRoot->keys[0] = newLeaf->keys[0];
 781      newRoot->children[0] = cursor;
 782      newRoot->children[1] = newLeaf;
 783      newRoot->keyCount = 1;
 784      *root = newRoot;
 785    }
 786    else {
 787      insertInternal(newLeaf->keys[0], NULL, findParent(*root, cursor), root);
 788    }
 789  }
 790}

 801  for (int i = 0; i < cursor->keyCount + 1; i++) {

 885/**

 887 *
 888 * @param pathFileRecipes File path to load the recipes.
 889 * @param pathFileIngredients File path to load the ingredients data.
 890 */
 1891void traverseRecipesBFS(const char* pathFileRecipes, const char* pathFileIngredients) {
 892  Recipe recipes[MAX_RECIPES];
 893  Ingredient* ingredientList = loadIngredientsFromFile(pathFileIngredients);
 894  int recipeCount = loadRecipesFromFile(pathFileRecipes, recipes, MAX_RECIPES);
 895
 896  if (recipeCount == 0) {
 897    printf("No recipes available."); enterToContinue(); return;
 898  }
 899
 900  std::queue<int> q;
 901  std::unordered_map<int, std::unordered_map<int, double>> ingredientUsage;
 902
 903  // Enqueue all recipes by their ID
 904  for (int i = 0; i < recipeCount; ++i) {
 905    q.push(i);
 906  }
 907
 908  // Traverse using BFS
 909  while (!q.empty()) {
 910    int recipeId = q.front();
 911    q.pop();
 912    Recipe currentRecipe = recipes[recipeId];
 913
